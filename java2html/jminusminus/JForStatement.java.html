2   package jminusminus;
3   
4   import static jminusminus.CLConstants.*;
5   
6   class JForStatement extends JStatement {
7   
8       /** initialization expression. */
9       private JStatement initialization;
10  
11      /** termination expression. */
12      private JExpression termination;
13  
14      /** increment expression. */
15      private JExpression increment;
16  
17      /** The body. */
18      private JStatement body;
19  
20      /**
21       * Construct an AST node for a while-statement given its line number, the
22       * test expression, and the body.
23       *
24       * @param line
25       *            line in which the while-statement occurs in the source file.
26       * @param condition
27       *            test expression.
28       * @param body
29       *            the body.
30       */
31  
32      public JForStatement(int line, JStatement initialization, JExpression termination, JExpression increment, JStatement body) {
33          super(line);
34          this.initialization = initialization;
35          this.termination = termination;
36          this.increment = increment;
37          this.body = body;
38      }
39  
40      /**
41       * Analysis involves analyzing the test, checking its type and analyzing the
42       * body statement.
43       *
44       * @param context
45       *            context in which names are resolved.
46       * @return the analyzed (and possibly rewritten) AST subtree.
47       */
48      public JForStatement analyze(Context context) {
49          this.initialization = (JStatement) initialization.analyze(context);
50          this.termination = (JExpression) termination.analyze(context);
51          this.increment = (JExpression) increment.analyze(context);
52          this.body = (JStatement) body.analyze(context);
53          return this;
54      }
55  
56      /**
57       * Generate code for the while loop.
58       *
59       * @param output
60       *            the code emitter (basically an abstraction for producing the
61       *            .class file).
62       */
63      public void codegen(CLEmitter output) {
64          // Need two labels
65          String test = output.createLabel();
66          String out = output.createLabel();
67  
68          // Codegen initializaton
69          initialization.codegen(output);
70  
71          // Branch out of the loop on the test condition
72          // being false
73          output.addLabel(test);
74          termination.codegen(output, out, false);
75  
76          // Codegen body
77          body.codegen(output);
78  
79          // Codegen increment
80          increment.codegen(output);
81  
82          // Unconditional jump back up to test
83          output.addBranchInstruction(GOTO, test);
84  
85          // The label below and outside the loop
86          output.addLabel(out);
87      }
88  
89      /**
90       * @inheritDoc
91       */
92      public void writeToStdOut(PrettyPrinter p) {
93          p.printf("<JForStatement line=\"%d\">\n", new Object[]{Integer.valueOf(this.line())});
94          p.indentRight();
95          p.printf("<IntializationExpression>\n", new Object[0]);
96          p.indentRight();
97          this.initialization.writeToStdOut(p);
98          p.indentLeft();
99          p.printf("</IntializationExpression>\n", new Object[0]);
100         p.printf("<TerminationExpression>\n", new Object[0]);
101         p.indentRight();
102         this.termination.writeToStdOut(p);
103         p.indentLeft();
104         p.printf("</TerminationExpression>\n", new Object[0]);
105         p.printf("<IncrementExpression>\n", new Object[0]);
106         p.indentRight();
107         this.increment.writeToStdOut(p);
108         p.indentLeft();
109         p.printf("</IncrementExpression>\n", new Object[0]);
110         p.printf("<Body>\n", new Object[0]);
111         p.indentRight();
112         this.body.writeToStdOut(p);
113         p.indentLeft();
114         p.printf("</Body>\n", new Object[0]);
115         p.indentLeft();
116         p.printf("</JForStatement>\n", new Object[0]);
117     }
118 }
