3   package jminusminus;
4   
5   import static jminusminus.CLConstants.*;
6   
7   /**
8    * The AST node for a do-while-statement.
9    */
10  
11  class JDoWhileStatement extends JStatement {
12  
13      /** Test expression. */
14      private JExpression condition;
15  
16      /** The body. */
17      private JStatement body;
18  
19      /**
20       * Construct an AST node for a do-while-statement given its line number, the
21       * test expression, and the body.
22       * 
23       * @param line
24       *            line in which the do-while-statement occurs in the source file.
25       * @param condition
26       *            test expression.
27       * @param body
28       *            the body.
29       */
30  
31      public JDoWhileStatement(int line, JExpression condition, JStatement body) {
32          super(line);
33          this.condition = condition;
34          this.body = body;
35      }
36  
37      /**
38       * Analysis involves analyzing the test, checking its type and analyzing the
39       * body statement.
40       * 
41       * @param context
42       *            context in which names are resolved.
43       * @return the analyzed (and possibly rewritten) AST subtree.
44       */
45  
46      public JDoWhileStatement analyze(Context context) {
47          condition = condition.analyze(context);
48          condition.type().mustMatchExpected(line(), Type.BOOLEAN);
49          body = (JStatement) body.analyze(context);
50          return this;
51      }
52  
53      /**
54       * Generate code for the do while loop.
55       * 
56       * @param output
57       *            the code emitter (basically an abstraction for producing the
58       *            .class file).
59       */
60  
61      public void codegen(CLEmitter output) {
62          // Add label to start of body
63          String start = output.createLabel();
64          output.addLabel(start);
65  
66          // Codegen body
67          body.codegen(output);
68  
69          // Branch to start of body if condition is true
70          condition.codegen(output, start, true);
71      }
72  
73      /**
74       * @inheritDoc
75       */
76  
77      public void writeToStdOut(PrettyPrinter p) {
78          p.printf("<JDoWhileStatement line=\"%d\">\n", line());
79          p.indentRight();
80          p.printf("<TestExpression>\n");
81          p.indentRight();
82          condition.writeToStdOut(p);
83          p.indentLeft();
84          p.printf("</TestExpression>\n");
85          p.printf("<Body>\n");
86          p.indentRight();
87          body.writeToStdOut(p);
88          p.indentLeft();
89          p.printf("</Body>\n");
90          p.indentLeft();
91          p.printf("</JDoWhileStatement>\n");
92      }
93  
94  }
